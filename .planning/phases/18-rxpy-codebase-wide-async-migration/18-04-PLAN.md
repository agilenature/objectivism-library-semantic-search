      """Async wrapper for occ_transition. Use in async def methods."""
      loop = asyncio.get_event_loop()
      result_future = loop.create_future()

      occ_transition(fn, max_attempts, base_delay).subscribe(
          on_next=lambda v: result_future.set_result(v) if not result_future.done() else None,
          on_error=lambda e: result_future.set_exception(e) if not result_future.done() else None,
          on_completed=lambda: None,
      )
      return await result_future

**Background (from MEMORY.md):**
The `poll_operation` function in client.py was fixed in Phase 16 (commit 64e2725):
the bug was `return operation` inside tenacity's `with attempt:` block exiting the
coroutine instead of retrying. Fix: `raise TryAgain` instead of `return`. The current
code uses `AsyncRetrying` + `TryAgain` correctly.

**Migration:**
Replace `AsyncRetrying` with `retry_when` using the confirmed contract from 18-01 spike
(Pattern 5). The key semantic: retry when `TryAgain` is raised (or a sentinel condition
is detected), stop when the operation is complete.

```python
# Before:
async for attempt in AsyncRetrying(stop=stop_after_attempt(MAX_POLL), wait=wait_fixed(POLL_INTERVAL)):
    with attempt:
        operation = await self._get_operation(operation_name)
        if not operation.done:
            raise TryAgain
        return operation

# After:
class _OperationNotDone(Exception):
    pass

async def _poll_once():
    operation = await self._get_operation(operation_name)
    if not operation.done:
        raise _OperationNotDone()  # sentinel for retry_when
    return operation

poll_future = asyncio.get_event_loop().create_future()

rx.defer(lambda: rx.from_future(asyncio.create_task(_poll_once()))).pipe(
    ops.retry_when(
        lambda errors: errors.pipe(
            ops.scan(lambda count, e: count + 1 if isinstance(e, _OperationNotDone) else count + MAX_POLL, 0),
            ops.flat_map(
                lambda n: rx.throw(TimeoutError(f"Operation did not complete after {MAX_POLL} polls"))
                if n >= MAX_POLL
                else rx.timer(POLL_INTERVAL)
            )
        )
    )
).subscribe(
    on_next=lambda v: poll_future.set_result(v) if not poll_future.done() else None,
    on_error=lambda e: poll_future.set_exception(e) if not poll_future.done() else None,
    on_completed=lambda: None,
)
result = await poll_future
```

Define `_OperationNotDone` as a private exception class in client.py.

**1. Semaphore+gather -> ops.map(factory).pipe(ops.merge(max_concurrent=N))**
Main upload fan-out. RxPY 3.x `ops.flat_map()` does NOT accept `max_concurrent`.
Use `ops.map` to produce deferred observables, then `ops.merge(max_concurrent=N)`:

```python
# Before:
sem = asyncio.Semaphore(MAX_CONCURRENT)
async def bounded(file_record):
    async with sem:
        return await upload(file_record)
results = await asyncio.gather(*[bounded(r) for r in file_records])

# After (using Future-based subscription to await results):
result_future = asyncio.get_event_loop().create_future()

rx.from_iterable(file_records).pipe(
    ops.map(lambda record: rx.defer(
        lambda record=record: rx.from_future(asyncio.create_task(upload(record)))
    )),
    ops.merge(max_concurrent=MAX_CONCURRENT),
    ops.to_list(),
).subscribe(
    on_next=lambda lst: result_future.set_result(lst),
    on_error=lambda e: result_future.set_exception(e),
    on_completed=lambda: None,
)
results = await result_future
```

Use `asyncio.create_task(upload(record))` — NOT `asyncio.run(upload(record))`.
`asyncio.run()` raises RuntimeError when a running event loop already exists.

**2. Staggered launches -> zip(rx.interval(1.0), source) + ops.map + ops.merge**

# Before:
for i, item in enumerate(items):

).subscribe(
    on_next=lambda v: collected.append(v),
    on_error=lambda e: result_future.set_exception(e) if not result_future.done() else None,

force_kill$ = rx.subject.Subject()       # take_until on active chains

    # Step 2: Await drain (wait for in-flight uploads to complete)
